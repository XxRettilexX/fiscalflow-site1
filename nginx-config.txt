server {
    listen 80;
    server_name fiscalflow.it www.fiscalflow.it;

    # Reindirizzamento forzato a HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name fiscalflow.it www.fiscalflow.it;

    # Configurazione SSL (Assicurati di avere i certificati)
    # ssl_certificate /etc/letsencrypt/live/fiscalflow.it/fullchain.pem;
    # ssl_certificate_key /etc/letsencrypt/live/fiscalflow.it/privkey.pem;
    
    # Protocolli TLS sicuri (Disabilita TLS 1.0 e 1.1)
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;

    # Protezione file sensibili
    location ~ /\.(?!well-known).* {
        deny all;
        access_log off;
        log_not_found off;
    }

    # Protezione specifica per file di configurazione e backup
    location ~* \.(env|git|json|lock|log|txt|yml|yaml|sh)$ {
        deny all;
        return 404;
    }

    # Rate Limiting (Protezione base contro Brute Force/Spam)
    # Aggiungi "limit_req_zone $binary_remote_addr zone=mylimit:10m rate=10r/s;" nel blocco http di nginx.conf principale
    # limit_req zone=mylimit burst=20 nodelay;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # Nascondi versione Nginx
        server_tokens off;
    }
}
